<script>

    // Sample database of registration numbers and their associated PDF links

    const policyData = {

        "TTUgWU1L": [

            "https://tempcover.motorcertificate.com/uploads/certificate_of_motor_insurance_M5%20YMK_122_20240811165111.pdf",

            "https://example.com/another-policy-document.pdf"

        ],

        "ABC123": [

            "https://example.com/sample-document.pdf"

        ]

    };


    const formContainer = document.getElementById('form-container');

    const policyContainer = document.getElementById('policy-container');

    const policyLinks = document.getElementById('policy-links');


    document.getElementById('validation-form').addEventListener('submit', function(e) {

        e.preventDefault();

        const registrationNumber = document.getElementById('registrationNumber').value;


        if (policyData[registrationNumber]) {

            showPolicyDetails(registrationNumber);

        } else {

            alert('Invalid Registration Number');

        }

    });


    function showPolicyDetails(registrationNumber) {

        policyLinks.innerHTML = '';

        policyData[registrationNumber].forEach(link => {

            const a = document.createElement('a');

            a.href = link;

            a.textContent = link;

            a.target = '_blank';

            policyLinks.appendChild(a);

        });

        formContainer.style.display = 'none';

        policyContainer.style.display = 'block';

    }


    function goBack() {

        formContainer.style.display = 'block';

        policyContainer.style.display = 'none';

    }

</script>
